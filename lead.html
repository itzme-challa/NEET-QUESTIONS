(async () => { // Function to wait for an element const waitForElement = (selector, timeout = 5000) => { return new Promise((resolve, reject) => { const element = document.querySelector(selector); if (element) return resolve(element); const startTime = Date.now(); const interval = setInterval(() => { const el = document.querySelector(selector); if (el) { clearInterval(interval); resolve(el); } else if (Date.now() - startTime > timeout) { clearInterval(interval); reject(new Error(`Timeout waiting for ${selector}`)); } }, 100); }); }; // Click Start button const startButton = await waitForElement('button#start-btn').catch(() => null); if (!startButton) { console.error('Start button not found'); return; } startButton.click(); // Navigate through up to 10 questions using Skip for (let i = 0; i < 10; i++) { await waitForElement('#question-container', 5000).catch(() => null); const skipButton = document.querySelector('#skip-btn'); if (!skipButton) { console.error('Skip button not found'); break; } skipButton.click(); await new Promise(resolve => setTimeout(resolve, 500)); } // Submit the quiz const submitButton = await waitForElement('#submit-btn[type="submit"]', 5000).catch(() => null); if (!submitButton) { console.error('Submit button not found'); return; } submitButton.click(); await new Promise(resolve => setTimeout(resolve, 2000)); // Wait for results page // Extract data from results table(s) const results = await waitForElement('table').then(() => { const sections = document.querySelectorAll('#headings'); const data = {}; sections.forEach(section => { const subject = section.querySelector('h3')?.textContent.replace('Section : ', '').trim() || 'Unknown'; const table = section.nextElementSibling; const rows = table.querySelectorAll('tbody tr'); const sectionData = []; for (const row of rows) { const questionNumber = row.querySelector('td:nth-child(1)')?.textContent.trim(); if (!questionNumber || questionNumber === 'N/A') continue; // Skip invalid question numbers const image = row.querySelector('.result-image')?.src || null; const answerText = row.querySelector('td:nth-child(3)')?.innerHTML || ''; const correctOptionMatch = answerText.match(/Correct Answer: ([A-D])/); const correctOption = correctOptionMatch ? correctOptionMatch[1] : 'N/A'; sectionData.push({ questionNumber: `Question ${questionNumber}`, image, correctOption }); } if (sectionData.length > 0) { data[subject] = sectionData; } }); // Ensure exactly 10 questions let totalQuestions = Object.values(data).reduce((sum, arr) => sum + arr.length, 0); if (totalQuestions < 10) { const placeholders = Array(10 - totalQuestions).fill().map((_, i) => ({ questionNumber: `Question ${totalQuestions + i + 1}`, image: null, correctOption: 'N/A' })); data['Unknown'] = (data['Unknown'] || []).concat(placeholders); } // Trim to exactly 10 questions const finalData = {}; let questionCount = 0; for (const subject in data) { const questions = data[subject]; const slice = questions.slice(0, Math.min(questions.length, 10 - questionCount)); if (slice.length > 0) { finalData[subject] = slice; questionCount += slice.length; } if (questionCount >= 10) break; } return finalData; }).catch(() => { console.error('Results table not found'); return {}; }); // Output JSON string const jsonString = JSON.stringify(results, null, 2); console.log('Copy the following JSON:'); console.log(jsonString); })();
